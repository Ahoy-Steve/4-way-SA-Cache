# 1. Quit any currently running simulation
quit -sim

# 2. Clear the transcript window
.main clear

# 3. Create the 'work' library (suppress error if it already exists)
vlib work

# 4. Compile the standard Verilog files
# (defs.vh is automatically included by the compiler)
# NOTE: cache_stats.v is now included so the statistics monitor is built.
vlog memory.v cache_controller.v cache_stats.v tb_cache.v

# 5. Start the simulator
# (-voptargs="+acc" prevents ModelSim from optimizing away internal signals so you can see them)
vsim -voptargs="+acc" work.tb_cache

# 6. Configure the Waveform Window
view wave
delete wave *

# --- Add System Signals ---
add wave -divider "System Signals"
add wave -color Gold /tb_cache/clock
add wave -color Gold /tb_cache/rst_n

# --- Add CPU Interface Signals ---
add wave -divider "CPU Interface"
add wave -radix hexadecimal /tb_cache/caddress
add wave /tb_cache/rden
add wave /tb_cache/wren
add wave -radix hexadecimal /tb_cache/cdin
add wave -color Green /tb_cache/hit
add wave -color Cyan -radix hexadecimal /tb_cache/cdout

# --- Add Internal Cache Debug Signals ---
add wave -divider "Internal Cache FSM"
add wave -color Yellow -radix unsigned /tb_cache/UUT_CACHE/current_state
add wave -radix hexadecimal /tb_cache/UUT_CACHE/active_index
add wave -radix hexadecimal /tb_cache/UUT_CACHE/active_tag
add wave -radix unsigned /tb_cache/UUT_CACHE/target_way

# --- Add Main Memory Signals ---
add wave -divider "Memory Interface"
add wave -radix hexadecimal /tb_cache/maddress
add wave /tb_cache/mrden
add wave /tb_cache/mwren

# --- Add Cache Statistics ---
add wave -divider "Statistics"
add wave -color Orange -radix unsigned /tb_cache/STATS/num_references
add wave -color Green  -radix unsigned /tb_cache/STATS/num_hits
add wave -color Red    -radix unsigned /tb_cache/STATS/num_misses

# 7. Run the simulation to the end
# (The numeric statistics report and the PASS/FAIL summary are printed
#  to the Transcript window by the testbench at the end of the run.)
run -all

# 8. Zoom the waveform to fit the entire simulation length
wave zoom full
